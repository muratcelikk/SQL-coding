------------------MANAGER AD-SOYAD-DEPARTMENT-MAAŞ ORTALAMASI----------

SELECT    
    DEP.DEPARTMENT_NAME,
        ROUND(
        (
            SELECT
                AVG( HRE.SALARY )
            FROM
                HR.EMPLOYEES HRE
            WHERE
                HRE.DEPARTMENT_ID = DEP.DEPARTMENT_ID
        ),
        2
    ) AVG_DEPARTMENT_SALARY,
    HRE.FIRST_NAME,
    HRE.LAST_NAME
FROM
    HR.DEPARTMENTS DEP,
    HR.EMPLOYEES HRE
WHERE
    HRE.EMPLOYEE_ID = DEP.MANAGER_ID
ORDER BY DEP.DEPARTMENT_NAME


--------------------MANAGER AD-SOYAD-DEPARTMENT-ALTINDA ÇALIŞAN KİŞİ SAYISI---------

SELECT    
    DEP.DEPARTMENT_NAME,
        ROUND(
        (
            SELECT
                COUNT(DEPARTMENT_ID)
            FROM
                HR.EMPLOYEES HRE
            WHERE
                HRE.DEPARTMENT_ID = DEP.DEPARTMENT_ID
        ),
        2
    ) AVG_DEPARTMENT_SALARY,
    HRE.FIRST_NAME,
    HRE.LAST_NAME
FROM
    HR.DEPARTMENTS DEP,
    HR.EMPLOYEES HRE
WHERE
    HRE.EMPLOYEE_ID = DEP.MANAGER_ID
ORDER BY DEP.DEPARTMENT_NAME

SELECT HRE.FIRST_NAME ||' '|| HRE.LAST_NAME AS ISIM_SOYAD, HRE.PHONE_NUMBER AS TELEFON, J.JOB_TITLE, D.DEPARTMENT_NAME AS DEPARTMAN_ADI

FROM HR.EMPLOYEES HRE

INNER JOIN HR.JOBS J ON J.JOB_ID = HRE.JOB_ID

INNER JOIN HR.DEPARTMENTS D ON D.DEPARTMENT_ID = HRE.DEPARTMENT_ID

-----------------------------------

 SELECT DISTINCT HRE.FIRST_NAME ||' '|| HRE.LAST_NAME AS ISIM_SOYAD, HRE.PHONE_NUMBER AS TELEFON, HRE.JOB_ID AS GUNCEL_İS, DEP.DEPARTMENT_NAME AS DEPARTMAN_ADI
 
 FROM HR.JOB_HISTORY H
 
 INNER JOIN HR.EMPLOYEES HRE ON HRE.EMPLOYEE_ID = H.EMPLOYEE_ID 
 
 INNER JOIN HR.DEPARTMENTS DEP ON DEP.DEPARTMENT_ID = H.DEPARTMENT_ID
